Monster Traits
==============

Generic
    hit_points = 10
    attack_power = 1
    attack_multiplier = 1
    defense = 1
    agility = 1
    spells = [] - list of dictionary keys. should they be strings or ints?
    party_weight = 1 - how many mobs this mob counts as for the purpose of
        assembling monster groups.
    gold_drop = 1 - how much gold per enemy defeated
    xp_drop = 1 - how much xp per enemy defeated
    item_dropped = "item" - Named quest-related item. Only special mobs have
        an "item_dropped" value. All others drop based on the general list for
        each tower level.
